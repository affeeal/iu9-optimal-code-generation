Program -> FuncDef
FuncDef -> IDENT ( FuncParams ) Scope
FuncParams -> FuncParams1 | empty
FuncParams1 -> IDENT | FuncParams1 , IDENT
Scope -> { Stmts }
Stmts -> Stmt | Stmts Stmt
Stmt -> VarDef | RetStmt | IfStmt | WhileStmt
VarDef -> IDENT = Expr ;
RetStmt -> return Expr ;
IfStmt -> if ( Expr ) Scope OptElse
OptElse -> else Scope | empty
WhileStmt -> while ( expr ) Scope
Expr -> ArithmExpr | Expr CmpOp ArithmExpr 
CmpOp -> == | != | < | <= | > | >=
ArithmExpr -> Term | ArithmExpr AddOp Term
AddOp -> + | - | ||
Term -> Factor | Term MulOp Factor
MulOp -> * | / | % | && 
Factor -> Atom | UnOp Atom
UnOp -> - | !
Atom -> IDENT | NUMBER | (Expr)

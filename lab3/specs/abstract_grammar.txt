Program -> FuncDefs
FuncDefs -> FuncDef | FuncDefs FuncDef
FuncDef -> FuncProto Scope
FuncProto -> IDENT ( FuncParamsOpt ) 
FuncParamsOpt -> FuncParams | empty
FuncParams -> IDENT | FuncParams , IDENT
Scope -> { Stmts }
Stmts -> Stmt | Stmts Stmt
Stmt -> AssignStmt | ReturnStmt | IfStmt | WhileStmt
AssignStmt -> IDENT = Expr ;
ReturnStmt -> return Expr ;
IfStmt -> if ( Expr ) Scope else Scope
WhileStmt -> while ( Expr ) Scope
Expr -> Expr BinOp Expr | UnOp Expr | IDENT | NUMBER | ( Expr )
BinOp -> == | != | < | > | <= | >= | +  | -  | || | *  | /  | % | &&
UnOp ->  - | !
